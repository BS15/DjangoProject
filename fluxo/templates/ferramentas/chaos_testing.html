3. Console Snippet — Injete direto no DevTools

Abra a página alvo em uma aba normal, pressione F12 → aba Console e cole o trecho abaixo. Ajuste NB_ATTACKS e DELAY_MS conforme necessário.

// ── Chaos Test: add_process.html ──────────────────────────────────────────── // Cole no console do DevTools enquanto estiver em /adicionar/ // Ajuste NB_ATTACKS e DELAY_MS conforme a intensidade desejada. (function () { const NB_ATTACKS = 500; // total de ataques aleatórios const DELAY_MS = 30; // ms entre cada ataque (menor = mais rápido) function loadGremlins(cb) { if (window.gremlins) return cb(); const s = document.createElement('script'); s.src = 'https://cdn.jsdelivr.net/npm/gremlins.js@2.2.0/dist/gremlins.min.js'; s.onload = cb; s.onerror = () => console.error('[chaos] Falha ao carregar gremlins.js'); document.head.appendChild(s); } function randomUnicode(len) { let s = ''; for (let i = 0; i < len; i++) s += String.fromCodePoint(Math.floor(Math.random() * 0xFFFF)); return s; } loadGremlins(function () { const g = window.gremlins; const horde = g.createHorde({ species: [ // 1. Clica em botões rapidamente → testa idempotência g.species.clicker({ clickTypes: ['click', 'dblclick'], canClick: (el) => ['BUTTON', 'A', 'INPUT', 'LABEL'].includes(el.tagName), log: true, }), // 2. Preenche inputs com strings Unicode aleatórias g.species.formFiller({ fillTypes: { 'text': () => randomUnicode(20), 'textarea': () => randomUnicode(80), 'email': () => randomUnicode(8) + '@' + randomUnicode(4) + '.com', 'number': () => String(Math.random() * 1e6 - 5e5 | 0), 'date': () => '2099-13-99', // propositalmente inválido }, log: true, }), // 3. Digita teclas aleatórias (inclui Enter — submete formulários) g.species.typer({ log: true }), ], mogwais: [ // Para automaticamente se um alert() aparecer (ex: confirmações do browser) g.mogwais.alert(), // Exibe FPS na tela para detectar travamentos g.mogwais.fps(), ], strategies: [ g.strategies.distribution({ nb: NB_ATTACKS, delay: DELAY_MS }), ], }); horde.after(function () { console.info('[chaos] Horde encerrada. Verifique os logs do Django por erros 500.'); }); console.info('[chaos] Soltando ' + NB_ATTACKS + ' gremlins em add_process…'); horde.unleash(); }); })();

4. Monitorar os Logs do Django em Tempo Real

Enquanto os gremlins correm, abra um terminal no servidor e use um dos comandos abaixo para enxergar exatamente quais views lançaram erros 500.

No terminal onde o servidor está rodando, cada requisição já aparece no stdout. Erros 500 são impressos com traceback completo. Filtre apenas os 500:

python manage.py runserver 2>&1 | grep -E "(500|ERROR|Traceback|Exception)"

Ou em paralelo, em outro terminal:

python manage.py runserver > /tmp/django.log 2>&1 & tail -f /tmp/django.log | grep -E "(500|ERROR|Traceback|Exception|Internal)"

Com DEBUG=True, o corpo da resposta 500 já exibe o traceback completo na própria página do iframe, então você também pode ler ali.

Se o servidor usa Gunicorn, redirecione os logs para um arquivo e faça tail:

gunicorn DjangoProject.wsgi:application \ --access-logfile /tmp/access.log \ --error-logfile /tmp/error.log \ --log-level debug & # Em outro terminal: tail -f /tmp/error.log | grep -E "(500|ERROR|Exception)"

O log de acesso registra POST /adicionar/ HTTP/1.1" 500 com timestamp — correlacione com os ataques pelo horário.

Adicione ao seu settings.py para gravar erros detalhados num arquivo:

LOGGING = { 'version': 1, 'disable_existing_loggers': False, 'formatters': { 'verbose': { 'format': '{levelname} {asctime} {module} {process:d} {thread:d} {message}', 'style': '{', }, }, 'handlers': { 'file_errors': { 'level': 'ERROR', 'class': 'logging.FileHandler', 'filename': BASE_DIR / 'logs' / 'django_errors.log', 'formatter': 'verbose', }, 'console': { 'class': 'logging.StreamHandler', 'formatter': 'verbose', }, }, 'loggers': { 'django': { 'handlers': ['file_errors', 'console'], 'level': 'ERROR', 'propagate': True, }, 'django.request': { 'handlers': ['file_errors', 'console'], 'level': 'ERROR', # captura cada erro 500 com traceback 'propagate': False, }, }, }

Depois, enquanto os gremlins atacam, observe em tempo real:

mkdir -p logs && tail -f logs/django_errors.log

Cada linha de ERROR corresponde a um view que gerou 500. O campo module indica exatamente qual view falhou.

  1. Abra a página alvo em uma aba e pressione F12.
  2. Vá para a aba Network e marque Preserve log.
  3. Filtre por status-code:500 (Chrome) ou use a caixa de filtro.
  4. Cole o console snippet e observe quais requisições retornam vermelho (500).
  5. Clique na requisição → aba Preview para ver o traceback Django (se DEBUG=True).

Combine com a Opção A/C para ter o traceback no servidor também, assim você sabe exatamente qual view e qual linha do código falhou.
